299 * This function calculates the orientation of a perceived object based on the angles provided in the `PerceivedObject` structure.
300 * The angles are converted to radians and used to set the roll, pitch, and yaw values of a `tf2::Quaternion` object.
301 * The resulting quaternion is then converted to a `gm::Quaternion` message and returned.
302 *
303 * @param object The `PerceivedObject` structure containing the angles of the perceived object.
304 * @return gm::Quaternion The orientation of the perceived object in radians from -pi to pi.
305 */
306inline gm::Quaternion getOrientationOfPerceivedObject(const PerceivedObject &object) {
307 if (!object.angles_is_present) throw std::invalid_argument("No angles present in PerceivedObject");
308 tf2::Quaternion q;
309 double roll{0}, pitch{0}, yaw{0};
310
311 if (object.angles.x_angle_is_present) {
312 roll = (((double(getCartesianAngle(object.angles.x_angle)) / 10.0) ) / 180.0) *
313 M_PI;
314 }
315 if (object.angles.y_angle_is_present) {
316 pitch = (((double(getCartesianAngle(object.angles.y_angle)) / 10.0) ) / 180.0) *
317 M_PI;
318 }
319 yaw = (((double(getCartesianAngle(object.angles.z_angle)) / 10.0)) / 180.0) *
320 M_PI;
321 // This wraps from -pi ti pi because setRPY only uses cos and sin of the angles
322 q.setRPY(roll, pitch, yaw);
323
324 return tf2::toMsg(q);
325}

333inline double getYawOfPerceivedObject(const PerceivedObject &object) {
334 gm::Quaternion orientation = getOrientationOfPerceivedObject(object);
335 tf2::Quaternion q(orientation.x, orientation.y, orientation.z, orientation.w);
336 double roll, pitch, yaw;
337 tf2::Matrix3x3(q).getRPY(roll, pitch, yaw);
338 return yaw;

425 *
426 * @param object PerceivedObject to get the Cartesian velocity from
427 * @return gm::Vector3 Cartesian velocity of the PerceivedObject in m/s (local coordinate system)
428 * @throws std::invalid_argument If the velocity is no cartesian velocity.
429 */
430inline gm::Vector3 getCartesianVelocityOfPerceivedObject(const PerceivedObject &object) {
431 if (!object.velocity_is_present) {
432 throw std::invalid_argument("No velocity present in PerceivedObject");
433 }
434 gm::Vector3 velocity;
435 if (object.velocity.choice == Velocity3dWithConfidence::CHOICE_POLAR_VELOCITY) {
436 double speed = getSpeed(object.velocity.polar_velocity.velocity_magnitude);
437 double angle = getCartesianAngle(object.velocity.polar_velocity.velocity_direction) * M_PI / 180.0 / 10.0; // convert to radians
438 velocity.x = speed * cos(angle);
439 velocity.y = speed * sin(angle);
440 if (object.velocity.polar_velocity.z_velocity_is_present) {
441 velocity.z = getVelocityComponent(object.velocity.cartesian_velocity.z_velocity);
442 }
443 } else if (object.velocity.choice == Velocity3dWithConfidence::CHOICE_CARTESIAN_VELOCITY) {
444 velocity.x = getVelocityComponent(object.velocity.cartesian_velocity.x_velocity);
445 velocity.y = getVelocityComponent(object.velocity.cartesian_velocity.y_velocity);
446 if (object.velocity.cartesian_velocity.z_velocity_is_present) {
447 velocity.z = getVelocityComponent(object.velocity.cartesian_velocity.z_velocity);
448 }
449 } else {
450 throw std::invalid_argument("Velocity is neither Polar nor Cartesian");
451 }
452 return velocity;
453}

461 * @throws std::invalid_argument If the velocity is no cartesian velocity.

463inline std::tuple<double, double, double> getCartesianVelocityConfidenceOfPerceivedObject(const PerceivedObject &object) {
464 if (!object.velocity_is_present) {
465 throw std::invalid_argument("No velocity present in PerceivedObject");
466 }
467 if (object.velocity.choice == Velocity3dWithConfidence::CHOICE_POLAR_VELOCITY) {
468 double speed_confidence = getSpeedConfidence(object.velocity.polar_velocity.velocity_magnitude) / etsi_its_msgs::ONE_D_GAUSSIAN_FACTOR;
469 double angle_confidence = getCartesianAngleConfidence(object.velocity.polar_velocity.velocity_direction) * M_PI / 180.0 / 10.0 / etsi_its_msgs::ONE_D_GAUSSIAN_FACTOR; // convert to radians
470 double speed = getSpeed(object.velocity.polar_velocity.velocity_magnitude);
471 double angle = getCartesianAngle(object.velocity.polar_velocity.velocity_direction) * M_PI / 180.0 / 10.0; // convert to radians
472 double lateral_confidence = speed * angle_confidence; // not exactly, but best approximation
473 double x_confidence = speed_confidence * cos(angle) * cos(angle)
474 + lateral_confidence * sin(angle) * sin(angle);
475 double y_confidence = speed_confidence * sin(angle) * sin(angle)
476 + lateral_confidence * cos(angle) * cos(angle);
477 // neglect xy covariance, as it is not present in the output of this function
478 double z_confidence = object.velocity.polar_velocity.z_velocity_is_present
479 ? getVelocityComponentConfidence(object.velocity.cartesian_velocity.z_velocity)
480 : std::numeric_limits<double>::infinity();
481 return std::make_tuple(x_confidence, y_confidence, z_confidence);
482 } else if (object.velocity.choice == Velocity3dWithConfidence::CHOICE_CARTESIAN_VELOCITY) {
483 double x_confidence = getVelocityComponentConfidence(object.velocity.cartesian_velocity.x_velocity);
484 double y_confidence = getVelocityComponentConfidence(object.velocity.cartesian_velocity.y_velocity);
485 double z_confidence = object.velocity.cartesian_velocity.z_velocity_is_present
486 ? getVelocityComponentConfidence(object.velocity.cartesian_velocity.z_velocity)
487 : std::numeric_limits<double>::infinity();
488 return std::make_tuple(x_confidence, y_confidence, z_confidence);
489 } else {
490 throw std::invalid_argument("Velocity is neither Polar nor Cartesian");
491 }
