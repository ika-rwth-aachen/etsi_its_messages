68inline gm::PointStamped getUTMPosition(const ReferencePosition& reference_position, int& zone, bool& northp){
69 gm::PointStamped utm_point;
70 double latitude = getLatitude(reference_position.latitude);
71 double longitude = getLongitude(reference_position.longitude);
72 utm_point.point.z = getAltitude(reference_position.altitude);
73 try {
74 GeographicLib::UTMUPS::Forward(latitude, longitude, zone, northp, utm_point.point.x, utm_point.point.y);
75 std::string hemisphere;
76 if(northp) hemisphere="N";
77 else hemisphere="S";
78 utm_point.header.frame_id="utm_"+std::to_string(zone)+hemisphere;
79 } catch (GeographicLib::GeographicErr& e) {
80 throw std::invalid_argument(e.what());
81 }
82 return utm_point;
83}
