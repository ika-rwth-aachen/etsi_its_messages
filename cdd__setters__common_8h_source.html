86inline void setAltitudeValue(AltitudeValue& altitude, const double value) {
87 int64_t alt_in_cm = (int64_t)std::round(value * 1e2);
88 if (alt_in_cm >= AltitudeValue::MIN && alt_in_cm <= AltitudeValue::MAX) {
89 altitude.value = alt_in_cm;
90 } else if (alt_in_cm < AltitudeValue::MIN) {
91 altitude.value = AltitudeValue::MIN;
92 } else if (alt_in_cm > AltitudeValue::MAX) {
93 altitude.value = AltitudeValue::MAX;
94 }
95}

184template <typename T>
185inline void setFromUTMPosition(T& reference_position, const gm::PointStamped& utm_position, const int zone,
186 const bool northp) {
187 std::string required_frame_prefix = "utm_";
188 if (utm_position.header.frame_id.rfind(required_frame_prefix, 0) != 0) {
189 throw std::invalid_argument("Frame-ID of UTM Position '" + utm_position.header.frame_id +
190 "' does not start with required prefix '" + required_frame_prefix + "'!");
191 }
192 double latitude, longitude;
193 try {
194 GeographicLib::UTMUPS::Reverse(zone, northp, utm_position.point.x, utm_position.point.y, latitude, longitude);
195 } catch (GeographicLib::GeographicErr& e) {
196 throw std::invalid_argument(e.what());
197 }
198 setReferencePosition(reference_position, latitude, longitude, utm_position.point.z);
199}

208template <typename T>
209inline void setBitString(T& bitstring, const std::vector<bool>& bits) {
210 // bit string size
211 const int bits_per_byte = 8;
212 const int n_bytes = (bits.size() - 1) / bits_per_byte + 1;
213 const int n_bits = n_bytes * bits_per_byte;
214
215 // init output
216 bitstring.bits_unused = n_bits - bits.size();
217 bitstring.value = std::vector<uint8_t>(n_bytes);
218
219 // loop over all bytes in reverse order
220 for (int byte_idx = n_bytes - 1; byte_idx >= 0; byte_idx--) {
221 // loop over bits in a byte
222 for (int bit_idx_in_byte = 0; bit_idx_in_byte < bits_per_byte; bit_idx_in_byte++) {
223 // map bit index in byte to bit index in total bitstring
224 int bit_idx = (n_bytes - byte_idx - 1) * bits_per_byte + bit_idx_in_byte;
225 if (byte_idx == 0 && bit_idx >= n_bits - bitstring.bits_unused) break;
226
227 // set bit in output bitstring appropriately
228 bitstring.value[byte_idx] |= bits[bit_idx] << bit_idx_in_byte;
229 }
230 }
