46inline void setGenerationDeltaTime(
47 GenerationDeltaTime& generation_delta_time, const uint64_t unix_nanosecs,
48 const uint16_t n_leap_seconds = etsi_its_msgs::LEAP_SECOND_INSERTIONS_SINCE_2004.rbegin()->second) {
49 TimestampIts t_its;
50 setTimestampITS(t_its, unix_nanosecs, n_leap_seconds);

54}

248inline void setExteriorLights(CAM& cam, const std::vector<bool>& exterior_lights) {
249 if (ExteriorLights::SIZE_BITS != exterior_lights.size()) {
250 throw std::invalid_argument("Vector has wrong size. (" + std::to_string(exterior_lights.size()) +
251 " != " + std::to_string(ExteriorLights::SIZE_BITS) + ")");
252 }
253 if (cam.cam.cam_parameters.low_frequency_container_is_present) {
254 if (cam.cam.cam_parameters.low_frequency_container.choice ==
255 LowFrequencyContainer::CHOICE_BASIC_VEHICLE_CONTAINER_LOW_FREQUENCY) {
256 setExteriorLights(
257 cam.cam.cam_parameters.low_frequency_container.basic_vehicle_container_low_frequency.exterior_lights,
258 exterior_lights);
259 } else {
260 throw std::invalid_argument("LowFrequencyContainer is not BASIC_VEHICLE_CONTAINER_LOW_FREQUENCY!");
261 }
262 } else {
263 throw std::invalid_argument("LowFrequencyContainer is not present!");
264 }
265}
